        PreparedStatement statement = null;
        ResultSet rs = null;
        int requestID = 0;

            /*Set up connection and Sql statement for Query */
            conn = getConnection();

                    + "from Request\n" 
                    + "order by RequestID desc";

            statement = conn.prepareStatement(sql);

            /*Assign data to an variable of Request*/
            while (rs.next()) {

            }

        } catch (SQLException ex) {
